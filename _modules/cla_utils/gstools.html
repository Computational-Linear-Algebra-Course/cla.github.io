[docs]def cgs(A):
    m, n = A.shape
    R = 0.0*zeros((n, n))
    Q = 1.0*A
    for j in range(n):
        v = A[:, j]
        if j>0:
            R[:j, j] = dot(Q[:,:j].T, A[:, j])
            v -= dot(R[:j, j], Q[:,:j].T)
        R[j, j] = linalg.norm(v)
        Q[:, j] = v/R[j, j]
    return Q, R




[docs]def mgs(A):
    m, n = A.shape
    R = 0.0*zeros((n, n))
    Q = 1.0*A
    for i in range(n):
        R[i, i] = linalg.norm(Q[:, i])
        Q[:, i] = Q[:, i]/R[i, i]
        R[i, i+1:] = dot(Q[:,i+1:].T, Q[:, i])
        Q[:, i+1:] -= outer(Q[:, i], R[i, i+1:])
    return Q, R
